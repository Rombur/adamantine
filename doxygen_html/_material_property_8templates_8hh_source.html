44 template <typename NumberType>
45 double MaterialProperty<dim>::get(
46  typename dealii::Triangulation<dim>::active_cell_iterator const &cell,
47  Property prop, dealii::LA::distributed::Vector<NumberType> const &) const
48 {
49  // TODO: For now, ignore field_state since we have a linear problem.
50  double value = 0.;
51  dealii::types::material_id material_id = cell->material_id();
52  unsigned int property = static_cast<unsigned int>(prop);
53 
54  double const mp_dof_index = get_dof_index(cell);
55 
56  for (unsigned int i = 0; i < _n_material_states; ++i)
57  {
58  // We cannot use operator[] because the function is constant.
59  auto const tmp = _properties.find(material_id);
60  ASSERT(tmp != _properties.end(), "Material not found.");
61  if ((tmp->second)[i][property] != nullptr)
62  value += _state[i][mp_dof_index] *
63  (tmp->second)[i][property]->value(dealii::Point<1>());
64  }
65 
66  return value;
67 }

69 template <int dim>
70 template <typename NumberType>
71 dealii::LA::distributed::Vector<NumberType>
72 MaterialProperty<dim>::enthalpy_to_temperature(
73  dealii::DoFHandler<dim> const &enthalpy_dof_handler,
74  dealii::LA::distributed::Vector<NumberType> const &enthalpy)
75 {
76  dealii::LA::distributed::Vector<NumberType> temperature(
77  enthalpy.get_partitioner());
78  dealii::LA::distributed::Vector<NumberType> dummy;
79 
80  update_state(enthalpy_dof_handler, enthalpy);
81 
82  unsigned int const dofs_per_cell =
83  enthalpy_dof_handler.get_fe().dofs_per_cell;
84  std::vector<dealii::types::global_dof_index> local_dof_indices(dofs_per_cell);
85  for (auto cell :
86  dealii::filter_iterators(enthalpy_dof_handler.active_cell_iterators(),
87  dealii::IteratorFilters::LocallyOwnedCell()))
88  {
89  cell->get_dof_indices(local_dof_indices);
90 
91  double const liquid_ratio = get_state_ratio(cell, MaterialState::liquid);
92  double const density = get(cell, Property::density, dummy);
93  double const specific_heat = get(cell, Property::specific_heat, dummy);
94  double const liquidus = get(cell, Property::liquidus, dummy);
95  double const solidus = get(cell, Property::solidus, dummy);
96  double const solidus_enthalpy = solidus * density * specific_heat;
97  double const latent_heat = get(cell, Property::latent_heat, dummy);
98  double const liquidus_enthalpy = solidus_enthalpy + latent_heat;
99 
100  NumberType enth_to_temp = [liquid_ratio, liquidus, solidus,
101  liquidus_enthalpy, solidus_enthalpy, density,
102  specific_heat](double const enthalpy)
103  {
104  if (liquid_ratio > 0.)
105  {
106  if (liquid_ratio == 1.)
107  {
108  return liquidus +
109  (enthalpy - liquidus_enthalpy) / (density * specific_heat);
110  }
111  else
112  {
113  return solidus +
114  (liquidus - solidus) * (enthalpy - solidus_enthalpy) /
115  (liquidus_enthalpy - solidus_enthalpy);
116  }
117  }
118  else
119  {
120  return enthalpy / (density * specific_heat);
121  }
122  };
123 
124  for (unsigned int i = 0; i < dofs_per_cell; ++i)
125  temperature[local_dof_indices[i]] =
126  enth_to_temp(enthalpy[local_dof_indices[i]]);
127  }
128 
129  return temperature;
130 }

141 template <int dim>
142 template <typename NumberType>
143 void MaterialProperty<dim>::update_state(
144  dealii::DoFHandler<dim> const &enthalpy_dof_handler,
145  dealii::LA::distributed::Vector<NumberType> const &enthalpy)
146 {
147  dealii::LA::distributed::Vector<NumberType> enthalpy_average =
148  compute_average_enthalpy(enthalpy_dof_handler, enthalpy);
149 
150  std::vector<dealii::types::global_dof_index> mp_dof(1);
151  for (auto cell :
152  dealii::filter_iterators(_mp_dof_handler.active_cell_iterators(),
153  dealii::IteratorFilters::LocallyOwnedCell()))
154  {
155  dealii::types::material_id material_id = cell->material_id();
156  auto const tmp = _properties.find(material_id);
157  ASSERT(tmp != _properties.end(), "Material not found.");
158 
159  // TODO check that assumption enthalpy of solidus powder = enthalpy of
160  // solidus solid
161  unsigned int constexpr liquid =
162  static_cast<unsigned int>(MaterialState::liquid);
163  unsigned int constexpr powder =
164  static_cast<unsigned int>(MaterialState::powder);
165  unsigned int constexpr solid =
166  static_cast<unsigned int>(MaterialState::solid);
167  unsigned int constexpr prop_latent_heat =
168  static_cast<unsigned int>(Property::latent_heat);
169  unsigned int constexpr prop_solidus =
170  static_cast<unsigned int>(Property::solidus);
171  unsigned int constexpr prop_density =
172  static_cast<unsigned int>(Property::density);
173  unsigned int constexpr prop_specific_heat =
174  static_cast<unsigned int>(Property::specific_heat);
175 
176  dealii::Point<1> const empty_pt;
177 
178  double const latent_heat =
179  ((tmp->second)[solid][prop_latent_heat] == nullptr)
180  ? std::numeric_limits<double>::max()
181  : (tmp->second)[solid][prop_latent_heat]->value(empty_pt);
182  double const solidus =
183  ((tmp->second)[solid][prop_solidus] == nullptr)
184  ? std::numeric_limits<double>::max()
185  : (tmp->second)[solid][prop_solidus]->value(empty_pt);
186  double const solidus_enthalpy =
187  solidus * (tmp->second)[solid][prop_density]->value(empty_pt) *
188  (tmp->second)[solid][prop_specific_heat]->value(empty_pt);
189  double const liquidus_enthalpy = solidus_enthalpy + latent_heat;
190  cell->get_dof_indices(mp_dof);
191  unsigned int const dof = mp_dof[0];
192 
193  // First determine the ratio of liquid.
194  double liquid_ratio = -1.;
195  double powder_ratio = -1.;
196  double solid_ratio = -1.;
197  if (enthalpy_average[dof] < solidus_enthalpy)
198  liquid_ratio = 0.;
199  else if (enthalpy_average[dof] > liquidus_enthalpy)
200  liquid_ratio = 1.;
201  else
202  liquid_ratio = (enthalpy_average[dof] - solidus_enthalpy) / latent_heat;
203  // Because the powder can only become liquid, the solid can only become
204  // liquid, and the liquid can only become solid, the ratio of powder can
205  // only decrease.
206  powder_ratio = std::min(1. - liquid_ratio, _state[powder][dof]);
207  // Use max to make sure that we don't create matter because of round-off.
208  solid_ratio = std::max(1 - liquid_ratio - powder_ratio, 0.);
209 
210  // Update the value
211  _state[liquid][dof] = liquid_ratio;
212  _state[powder][dof] = powder_ratio;
213  _state[solid][dof] = solid_ratio;
214  }
215 }

231 template <int dim>
232 void MaterialProperty<dim>::fill_properties(
233  boost::property_tree::ptree const &database)
234 {
235  std::array<std::string, _n_material_states> material_state = {
236  {"powder", "solid", "liquid"}};
237  std::array<std::string, _n_properties> properties = {
238  {"density", "latent_heat", "liquidus", "solidus", "specific_heat",
239  "thermal_conductivity"}};
240 
241  unsigned int const n_materials = database.get<unsigned int>("n_materials");
242  dealii::types::material_id next_material_id = 0;
243  for (unsigned int i = 0; i < n_materials; ++i)
244  {
245  // Try to find the material_id by checking every possible number.
246  dealii::types::material_id material_id = next_material_id;
247  while (material_id < dealii::numbers::invalid_material_id)
248  {
249  // If the child exists, exit the loop.
250  if (database.count("material_" + std::to_string(material_id)) != 0)
251  break;
252 
253  ++material_id;
254  }
255  ASSERT_THROW(material_id != dealii::numbers::invalid_material_id,
256  "Invalid material ID. Choose a smaller number");
257  // Update the next possible material id
258  ++next_material_id;
259 
260  // Get the material property tree.
261  std::string variable = "T";
262  boost::property_tree::ptree const &material_database =
263  database.get_child("material_" + std::to_string(material_id));
264  // For each material, loop over the possible states.
265  bool valid_state = false;
266  for (unsigned int state = 0; state < _n_material_states; ++state)
267  {
268  // The state may or may not exist for the material.
269  boost::optional<boost::property_tree::ptree const &> state_database =
270  material_database.get_child_optional(material_state[state]);
271  if (state_database)
272  {
273  valid_state = true;
274  // For each state, loop over the possible properties.
275  for (unsigned int p = 0; p < _n_properties; ++p)
276  {
277  // The property may or may not exist for that state
278  boost::optional<std::string> const property =
279  state_database.get().get_optional<std::string>(properties[p]);
280  // If the property exists, put it in the map. If the property does not
281  // exist, we have a nullptr.
282  if (property)
283  {
284  _properties[material_id][state][p] =
285  std::make_unique<dealii::FunctionParser<1>>(1);
286  _properties[material_id][state][p]->initialize(
287  variable, property.get(), std::map<std::string, double>());
288  }
289  }
290  }
291  }
292  // Check that there is at least one valid MaterialState
293  ASSERT_THROW(
294  valid_state == true,
295  "Material without any valid state (solid, powder, or liquid).");
296 
297  // Check for the properties that are associated to a material but that
298  // are independent of an individual state. These properties are duplicated
299  // for every state.
300  for (unsigned int p = 0; p < _n_properties; ++p)
301  {
302  // The property may or may not exist for that state
303  boost::optional<std::string> const property =
304  material_database.get_optional<std::string>(properties[p]);
305  // If the property exists, put it in the map. If the property does not
306  // exist, we have a nullptr.
307  if (property)
308  {
309  for (unsigned int state = 0; state < _n_material_states; ++state)
310  {
311  _properties[material_id][state][p] =
312  std::make_unique<dealii::FunctionParser<1>>(1);
313  _properties[material_id][state][p]->initialize(
314  variable, property.get(), std::map<std::string, double>());
315  }
316  }
317  }
318  }

321 template <int dim>
322 void MaterialProperty<dim>::compute_constants()
323 {
324  unsigned int constexpr liquid =
325  static_cast<unsigned int>(MaterialState::liquid);
326  unsigned int constexpr solid =
327  static_cast<unsigned int>(MaterialState::solid);
328  unsigned int constexpr prop_latent_heat =
329  static_cast<unsigned int>(Property::latent_heat);
330  unsigned int constexpr solidus = static_cast<unsigned int>(Property::solidus);
331  unsigned int constexpr liquidus =
332  static_cast<unsigned int>(Property::liquidus);
333  unsigned int constexpr density = static_cast<unsigned int>(Property::density);
334  unsigned int constexpr specific_heat =
335  static_cast<unsigned int>(Property::specific_heat);
336 
337  dealii::Point<1> const empty_pt;
338 
339  for (auto prop = _properties.begin(); prop != _properties.end(); ++prop)
340  {
341  dealii::types::material_id const material_id = prop->first;
342  bool const liquidus_exist =
343  (prop->second[solid][liquidus] == nullptr) ? false : true;
344  bool const solidus_exist =
345  (prop->second[solid][solidus] == nullptr) ? false : true;
346  bool const latent_heat_exist =
347  (prop->second[solid][prop_latent_heat] == nullptr) ? false : true;
348  bool const density_exist =
349  (prop->second[solid][density] == nullptr) ? false : true;
350  bool const specific_heat_exist =
351  (prop->second[solid][specific_heat] == nullptr) ? false : true;
352 
353  if (liquidus_exist && solidus_exist && latent_heat_exist)
354  _mushy_alpha[material_id] =
355  (prop->second[solid][liquidus]->value(empty_pt) -
356  prop->second[solid][solidus]->value(empty_pt)) /
357  prop->second[solid][prop_latent_heat]->value(empty_pt);
358 
359  if (liquidus_exist && solidus_exist && density_exist && specific_heat_exist)
360  {
361  double const solidus_enthalpy =
362  prop->second[solid][solidus]->value(empty_pt) *
363  prop->second[solid][density]->value(empty_pt) *
364  prop->second[solid][specific_heat]->value(empty_pt);
365  _mushy_beta[material_id] =
366  -solidus_enthalpy /
367  prop->second[solid][prop_latent_heat]->value(empty_pt) *
368  (prop->second[solid][liquidus]->value(empty_pt) -
369  prop->second[solid][solidus]->value(empty_pt)) +
370  prop->second[solid][solidus]->value(empty_pt);
371  // TODO this is true only if density and heat capacity are independent of
372  // the temperature
373  _liquid_beta[material_id] =
374  -(solidus_enthalpy +
375  prop->second[liquid][prop_latent_heat]->value(empty_pt)) /
376  (prop->second[liquid][density]->value(empty_pt) *
377  prop->second[liquid][specific_heat]->value(empty_pt)) +
378  prop->second[liquid][liquidus]->value(empty_pt);
379  }
380  }
381 }

383 template <int dim>
384 template <typename NumberType>
385 dealii::LA::distributed::Vector<NumberType>
386 MaterialProperty<dim>::compute_average_enthalpy(
387  dealii::DoFHandler<dim> const &enthalpy_dof_handler,
388  dealii::LA::distributed::Vector<NumberType> const &enthalpy) const
389 {
390  // TODO: this should probably done in a matrix-free fashion.
391  // The triangulation is the same for both DoFHandler
392  dealii::LA::distributed::Vector<NumberType> enthalpy_average(
393  _mp_dof_handler.locally_owned_dofs(), enthalpy.get_mpi_communicator());
394  enthalpy_average = 0.;
395  auto mp_cell = _mp_dof_handler.begin_active();
396  auto mp_end_cell = _mp_dof_handler.end();
397  auto enth_cell = enthalpy_dof_handler.begin_active();
398  dealii::FiniteElement<dim> const &fe = enthalpy_dof_handler.get_fe();
399  // We can use a lower degree of quadrature since we are projecting on a
400  // piecewise constant space
401  dealii::QGauss<dim> quadrature(fe.degree);
402  dealii::FEValues<dim> fe_values(
403  fe, quadrature, dealii::UpdateFlags::update_values |
404  dealii::UpdateFlags::update_quadrature_points |
405  dealii::UpdateFlags::update_JxW_values);
406  unsigned int const dofs_per_cell = fe.dofs_per_cell;
407  std::vector<dealii::types::global_dof_index> mp_dof_indices(1);
408  std::vector<dealii::types::global_dof_index> enth_dof_indices(dofs_per_cell);
409  unsigned int const n_q_points = quadrature.size();
410  for (; mp_cell != mp_end_cell; ++enth_cell, ++mp_cell)
411  {
412  fe_values.reinit(enth_cell);
413  mp_cell->get_dof_indices(mp_dof_indices);
414  dealii::types::global_dof_index const mp_dof_index = mp_dof_indices[0];
415  enth_cell->get_dof_indices(enth_dof_indices);
416  double area = 0.;
417  for (unsigned int q = 0; q < n_q_points; ++q)
418  for (unsigned int i = 0; i < dofs_per_cell; ++i)
419  {
420  area += fe_values.shape_value(i, q) * fe_values.JxW(q);
421  enthalpy_average[mp_dof_index] += fe_values.shape_value(i, q) *
422  enthalpy[enth_dof_indices[i]] *
423  fe_values.JxW(q);
424  }
425  enthalpy_average[mp_dof_index] /= area;
426  }
427 
428  return enthalpy_average;
429 }
