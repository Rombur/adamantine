36  boost::property_tree::ptree const &source_database =
37  database.get_child("sources");

40  for (unsigned int i = 0; i < n_beams; ++i)
41  {
42  boost::property_tree::ptree const &beam_database =
43  source_database.get_child("beam_" + std::to_string(i));
44  _electron_beams[i] = std::make_unique<ElectronBeam<dim>>(beam_database);

51  std::make_shared<ThermalOperator<dim, fe_degree, NumberType>>(
52  communicator, _material_properties);
53 
54  // Create the time stepping scheme
55  boost::property_tree::ptree const &time_stepping_database =
56  database.get_child("time_stepping");

58  std::transform(method.begin(), method.end(), method.begin(),
59  [](unsigned char c)
60  {
61  return std::tolower(c);
62  });

64  _time_stepping =
65  std::make_unique<dealii::TimeStepping::ExplicitRungeKutta<LA_Vector>>(
66  dealii::TimeStepping::FORWARD_EULER);
67  else if (method.compare("rk_third_order") == 0)
68  _time_stepping =
69  std::make_unique<dealii::TimeStepping::ExplicitRungeKutta<LA_Vector>>(
70  dealii::TimeStepping::RK_THIRD_ORDER);
71  else if (method.compare("rk_fourth_order") == 0)
72  _time_stepping =
73  std::make_unique<dealii::TimeStepping::ExplicitRungeKutta<LA_Vector>>(
74  dealii::TimeStepping::RK_CLASSIC_FOURTH_ORDER);
75  else if (method.compare("heun_euler") == 0)
76  {
77  _time_stepping = std::make_unique<
78  dealii::TimeStepping::EmbeddedExplicitRungeKutta<LA_Vector>>(
79  dealii::TimeStepping::HEUN_EULER);
80  _embedded_method = true;
81  }
82  else if (method.compare("bogacki_shampine") == 0)
83  {
84  _time_stepping = std::make_unique<
85  dealii::TimeStepping::EmbeddedExplicitRungeKutta<LA_Vector>>(
86  dealii::TimeStepping::BOGACKI_SHAMPINE);
87  _embedded_method = true;
88  }
89  else if (method.compare("dopri") == 0)
90  {
91  _time_stepping = std::make_unique<
92  dealii::TimeStepping::EmbeddedExplicitRungeKutta<LA_Vector>>(
93  dealii::TimeStepping::DOPRI);
94  _embedded_method = true;
95  }
96  else if (method.compare("fehlberg") == 0)
97  {
98  _time_stepping = std::make_unique<
99  dealii::TimeStepping::EmbeddedExplicitRungeKutta<LA_Vector>>(
100  dealii::TimeStepping::FEHLBERG);
101  _embedded_method = true;
102  }

108  _embedded_method = true;
109  }

122  _implicit_method = true;
123  }

125  {
126  _time_stepping =
127  std::make_unique<dealii::TimeStepping::ImplicitRungeKutta<LA_Vector>>(
128  dealii::TimeStepping::CRANK_NICOLSON);
129  _implicit_method = true;
130  }
131  else if (method.compare("sdirk2") == 0)
132  {
133  _time_stepping =
134  std::make_unique<dealii::TimeStepping::ImplicitRungeKutta<LA_Vector>>(
135  dealii::TimeStepping::SDIRK_TWO_STAGES);
136  _implicit_method = true;
137  }
138 
139  if (_embedded_method == true)
140  {
141  double coarsen_param =
142  time_stepping_database.get("coarsening_parameter", 1.2);
143  double refine_param = time_stepping_database.get("refining_parameter", 0.8);
144  double min_delta = time_stepping_database.get("min_time_step", 1e-14);

151  dealii::TimeStepping::EmbeddedExplicitRungeKutta<LA_Vector> *>(

154  min_delta, max_delta,
155  refine_tol, coarsen_tol);
156  }
157

160  if (_implicit_method == true)

207 template <int dim, int fe_degree, typename NumberType, typename QuadratureType>
208 double ThermalPhysics<dim, fe_degree, NumberType, QuadratureType>::
209  evolve_one_time_step(double t, double delta_t,
210  dealii::LA::distributed::Vector<NumberType> &solution)
211 {
212  double time = _time_stepping->evolve_one_time_step(
213  std::bind(&ThermalPhysics<dim, fe_degree, NumberType,
214  QuadratureType>::evaluate_thermal_physics,
215  this, std::placeholders::_1, std::placeholders::_2),
216  std::bind(&ThermalPhysics<dim, fe_degree, NumberType,
217  QuadratureType>::id_minus_tau_J_inverse,
218  this, std::placeholders::_1, std::placeholders::_2,
219  std::placeholders::_3),
220  t, delta_t, solution);
221 
222  // If the method is embedded, get the next time step. Otherwise, just use the
223  // current time step.
224  if (_embedded_method == false)
225  _delta_t_guess = delta_t;
226  else
227  {
228  dealii::TimeStepping::EmbeddedExplicitRungeKutta<LA_Vector> *embedded_rk =
229  static_cast<
230  dealii::TimeStepping::EmbeddedExplicitRungeKutta<LA_Vector> *>(
231  _time_stepping.get());
232  _delta_t_guess = embedded_rk->get_status().delta_t_guess;
233  }
234 
235  // Return the time at the end of the time step. This may be different than
236  // t+delta_t for embedded methods.
237  return time;
238 }

240 template <int dim, int fe_degree, typename NumberType, typename QuadratureType>
241 dealii::LA::distributed::Vector<NumberType>
242 ThermalPhysics<dim, fe_degree, NumberType, QuadratureType>::
243  evaluate_thermal_physics(
244  double const t,
245  dealii::LA::distributed::Vector<NumberType> const &y) const
246 {
247  LA_Vector value(y.get_partitioner());
248  value = 0.;
249 
250  // Compute the source term.
251  for (auto &beam : _electron_beams)
252  beam->set_time(t);
253  dealii::QGauss<dim> source_quadrature(fe_degree + 1);
254  dealii::FEValues<dim> fe_values(_fe, source_quadrature,
255  dealii::update_quadrature_points |
256  dealii::update_values |
257  dealii::update_JxW_values);
258  unsigned int const dofs_per_cell = _fe.dofs_per_cell;
259  unsigned int const n_q_points = source_quadrature.size();
260  std::vector<dealii::types::global_dof_index> local_dof_indices(dofs_per_cell);
261  dealii::Vector<NumberType> cell_source(dofs_per_cell);
262 
263  for (auto cell :
264  dealii::filter_iterators(_dof_handler.active_cell_iterators(),
265  dealii::IteratorFilters::LocallyOwnedCell()))
266  {
267  cell_source = 0.;
268  fe_values.reinit(cell);
269  NumberType const rho_cp =
270  _material_properties->get<dim, NumberType>(cell, Property::density, y) *
271  _material_properties->get<dim, NumberType>(cell,
272  Property::specific_heat, y);
273 
274  for (unsigned int i = 0; i < dofs_per_cell; ++i)
275  {
276  for (unsigned int q = 0; q < n_q_points; ++q)
277  {
278  double source = 0.;
279  dealii::Point<dim> const &q_point = fe_values.quadrature_point(q);
280  for (auto &beam : _electron_beams)
281  source += beam->value(q_point);
282  source /= rho_cp;
283 
284  cell_source[i] +=
285  source * fe_values.shape_value(i, q) * fe_values.JxW(q);
286  }
287  }
288  cell->get_dof_indices(local_dof_indices);
289  _constraint_matrix.distribute_local_to_global(cell_source,
290  local_dof_indices, value);
291  }
292 
293  // Apply the Thermal Operator.
294  _thermal_operator->vmult_add(value, y);
295 
296  // Multiply by the inverse of the mass matrix.
297  value.scale(*_thermal_operator->get_inverse_mass_matrix());
298 
299  return value;
300 }

302 template <int dim, int fe_degree, typename NumberType, typename QuadratureType>
303 dealii::LA::distributed::Vector<NumberType>
304 ThermalPhysics<dim, fe_degree, NumberType, QuadratureType>::
305  id_minus_tau_J_inverse(
306  double const /*t*/, double const tau,
307  dealii::LA::distributed::Vector<NumberType> const &y) const
308 {
309  _implicit_operator->set_tau(tau);
310  dealii::LA::distributed::Vector<NumberType> solution(y.get_partitioner());
311 
312  // TODO Add a geometric multigrid preconditioner.
313  dealii::PreconditionIdentity preconditioner;
314

316  // TODO Check why we need to use GMRES instead of CG and why it is so hard to

322  solver_control, additional_data);
