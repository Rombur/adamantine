41 template <int dim, int fe_degree, typename NumberType>
42 void ThermalOperator<dim, fe_degree, NumberType>::reinit(
43  dealii::DoFHandler<dim> const &dof_handler,
44  dealii::ConstraintMatrix const &constraint_matrix)
45 {
46  // Compute the inverse of the mass matrix
47  dealii::QGaussLobatto<1> mass_matrix_quad(fe_degree + 1);
48  dealii::MatrixFree<dim, NumberType> mass_matrix_free;
49  mass_matrix_free.reinit(dof_handler, constraint_matrix, mass_matrix_quad,
50  _matrix_free_data);
51  mass_matrix_free.initialize_dof_vector(*_inverse_mass_matrix);
52  dealii::VectorizedArray<NumberType> one =
53  dealii::make_vectorized_array(static_cast<NumberType>(1.));
54  dealii::FEEvaluation<dim, fe_degree, fe_degree + 1, 1, NumberType> fe_eval(
55  mass_matrix_free);
56  unsigned int const n_q_points = fe_eval.n_q_points;
57  for (unsigned int cell = 0; cell < mass_matrix_free.n_macro_cells(); ++cell)
58  {
59  fe_eval.reinit(cell);
60  for (unsigned int q = 0; q < n_q_points; ++q)
61  fe_eval.submit_value(one, q);
62  fe_eval.integrate(true, false);
63  fe_eval.distribute_local_to_global(*_inverse_mass_matrix);
64  }
65  _inverse_mass_matrix->compress(dealii::VectorOperation::add);
66  unsigned int const local_size = _inverse_mass_matrix->local_size();
67  for (unsigned int k = 0; k < local_size; ++k)
68  {
69  if (_inverse_mass_matrix->local_element(k) > 1e-15)
70  _inverse_mass_matrix->local_element(k) =
71  1. / _inverse_mass_matrix->local_element(k);
72  else
73  _inverse_mass_matrix->local_element(k) = 0.;
74  }
75 }

102 template <int dim, int fe_degree, typename NumberType>
103 void ThermalOperator<dim, fe_degree, NumberType>::vmult_add(
104  dealii::LA::distributed::Vector<NumberType> &dst,
105  dealii::LA::distributed::Vector<NumberType> const &src) const
106 {
107  // Execute the matrix-free matrix-vector multiplication
108  _matrix_free.cell_loop(&ThermalOperator::local_apply, this, dst, src);
109 
110  // Because cell_loop resolves the constraints, the constrained dofs are not
111  // called they stay at zero. Thus, we need to force the value on the
112  // constrained dofs by hand. The variable scaling is used so that we get the
113  // right order of magnitude.
114  // TODO: for now the value of scaling is set to 1
115  NumberType const scaling = 1.;
116  std::vector<unsigned int> const &constrained_dofs =
117  _matrix_free.get_constrained_dofs();
118  for (auto &dof : constrained_dofs)
119  dst.local_element(dof) += scaling * src.local_element(dof);
120 }

131 template <int dim, int fe_degree, typename NumberType>
132 void ThermalOperator<dim, fe_degree, NumberType>::local_apply(
133  dealii::MatrixFree<dim, NumberType> const &data,
134  dealii::LA::distributed::Vector<NumberType> &dst,
135  dealii::LA::distributed::Vector<NumberType> const &src,
136  std::pair<unsigned int, unsigned int> const &cell_range) const
137 {
138  dealii::FEEvaluation<dim, fe_degree, fe_degree + 1, 1, NumberType> fe_eval(
139  data);
140  dealii::Tensor<1, dim> unit_tensor;
141  for (unsigned int i = 0; i < dim; ++i)
142  unit_tensor[i] = 1.;
143 
144  // Loop over the "cells". Note that we don't really work on a cell but on a
145  // set of quadrature point.
146  for (unsigned int cell = cell_range.first; cell < cell_range.second; ++cell)
147  {
148  // Reinit fe_eval on the current cell
149  fe_eval.reinit(cell);
150  // Store in a local vector the local values of src
151  fe_eval.read_dof_values(src);
152  // Evaluate the only the function gradients on the reference cell
153  fe_eval.evaluate(false, true);
154  // Apply the Jacobian of the transformation, multiply by the variable
155  // coefficients and the quadrature points
156  for (unsigned int q = 0; q < fe_eval.n_q_points; ++q)
157  fe_eval.submit_gradient(-_thermal_conductivity(cell, q) *
158  (fe_eval.get_gradient(q) * _alpha(cell, q) +
159  unit_tensor * _beta(cell, q)),
160  q);
161  // Sum over the quadrature points.
162  fe_eval.integrate(false, true);
163  fe_eval.distribute_local_to_global(dst);
164  }
165 }

167 template <int dim, int fe_degree, typename NumberType>
168 void ThermalOperator<dim, fe_degree, NumberType>::evaluate_material_properties(
169  dealii::LA::distributed::Vector<NumberType> const &state)
170 {
171  // Update the state of the materials
172  _material_properties->update_state(_matrix_free.get_dof_handler(), state);
173 
174  unsigned int const n_cells = _matrix_free.n_macro_cells();
175  dealii::FEEvaluation<dim, fe_degree, fe_degree + 1, 1, NumberType> fe_eval(
176  _matrix_free);
177  _alpha.reinit(n_cells, fe_eval.n_q_points);
178  _beta.reinit(n_cells, fe_eval.n_q_points);
179  _thermal_conductivity.reinit(n_cells, fe_eval.n_q_points);
180  for (unsigned int cell = 0; cell < n_cells; ++cell)
181  for (unsigned int q = 0; q < fe_eval.n_q_points; ++q)
182  for (unsigned int i = 0; i < _matrix_free.n_components_filled(cell); ++i)
183  {
184  typename dealii::DoFHandler<dim>::cell_iterator cell_it =
185  _matrix_free.get_cell_iterator(cell, i);
186  // Cast to Triangulation<dim>::cell_iterator to access the material_id
187  typename dealii::Triangulation<dim>::active_cell_iterator cell_tria(
188  cell_it);
189 
190  _thermal_conductivity(cell, q)[i] = _material_properties->get(
191  cell_tria, Property::thermal_conductivity, state);
192 
193  double liquid_ratio = _material_properties->get_state_ratio(
194  cell_tria, MaterialState::liquid);
195 
196  // If there is less than 1e-13% of liquid, assume that there is no
197  // liquid.
198  if (liquid_ratio < 1e-15)
199  {
200  _alpha(cell, q)[i] =
201  1. /
202  (_material_properties->get(cell_tria, Property::density, state) *
203  _material_properties->get(cell_tria, Property::specific_heat,
204  state));
205  }
206  else
207  {
208  // If there is less than 1e-13% of solid, assume that there is no
209  // solid. Otherwise, we have a mix of liquid and solid (mushy zone).
210  if (liquid_ratio > (1 - 1e-15))
211  {
212  _alpha(cell, q)[i] =
213  1. / (_material_properties->get(cell_tria, Property::density,
214  state) *
215  _material_properties->get(
216  cell_tria, Property::specific_heat, state));
217  _beta(cell, q)[i] =
218  _material_properties->get_liquid_beta(cell_tria);
219  }
220  else
221  {
222  _alpha(cell, q)[i] =
223  _material_properties->get_mushy_alpha(cell_tria);
224  _beta(cell, q)[i] = _material_properties->get_mushy_beta(cell_tria);
225  }
226  }
227  }
228 }
229 }
