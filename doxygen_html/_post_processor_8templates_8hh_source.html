29 template <int dim>
30 void PostProcessor<dim>::output_pvtu(
31  unsigned int cycle, unsigned int time_step, double time,
32  dealii::LA::distributed::Vector<double> const &solution)
33 {
34  // Add the DoFHandler and the enthalpy.
35  _data_out.clear();
36  _data_out.attach_dof_handler(_dof_handler);
37  solution.update_ghost_values();
38  _data_out.add_data_vector(solution, "enthalpy");
39 
40  // Add the temperature.
41  dealii::LA::distributed::Vector<double> temperature =
42  compute_temperature(solution);
43  _data_out.add_data_vector(temperature, "temperature");
44 
45  // Add MaterialState ratio
46  std::array<dealii::LA::distributed::Vector<double>,
47  static_cast<unsigned int>(MaterialState::SIZE)> state =
48  _material_properties->get_state();
49  _data_out.add_data_vector(
50  state[static_cast<unsigned int>(MaterialState::powder)], "powder");
51  _data_out.add_data_vector(
52  state[static_cast<unsigned int>(MaterialState::liquid)], "liquid");
53  _data_out.add_data_vector(
54  state[static_cast<unsigned int>(MaterialState::solid)], "solid");
55 
56  // Add the subdomain IDs.
57  dealii::types::subdomain_id subdomain_id =
58  _dof_handler.get_triangulation().locally_owned_subdomain();
59  dealii::Vector<float> subdomain(
60  _dof_handler.get_triangulation().n_active_cells());
61  for (unsigned int i = 0; i < subdomain.size(); ++i)
62  subdomain[i] = subdomain_id;
63  _data_out.add_data_vector(subdomain, "subdomain");
64 
65  // Output the data.
66  _data_out.build_patches();
67  std::string local_filename =
68  _filename + "." + dealii::Utilities::int_to_string(cycle, 2) + "." +
69  dealii::Utilities::int_to_string(time_step, 6) + "." +
70  dealii::Utilities::int_to_string(subdomain_id, 6);
71  std::ofstream output((local_filename + ".vtu").c_str());
72  dealii::DataOutBase::VtkFlags flags(time, cycle);
73  _data_out.set_flags(flags);
74  _data_out.write_vtu(output);
75 
76  // Output the master record.
77  if (_communicator.rank() == 0)
78  {
79  std::vector<std::string> filenames;
80  for (int i = 0; i < _communicator.size(); ++i)
81  {
82  std::string local_name =
83  _filename + "." + dealii::Utilities::int_to_string(cycle, 2) + "." +
84  dealii::Utilities::int_to_string(time_step, 6) + "." +
85  dealii::Utilities::int_to_string(i, 6) + ".vtu";
86  filenames.push_back(local_name);
87  }
88  std::string master_filename =
89  _filename + "." + dealii::Utilities::int_to_string(cycle, 2) + "." +
90  dealii::Utilities::int_to_string(time_step, 6) + ".pvtu";
91  std::ofstream master_output(master_filename.c_str());
92  _data_out.write_pvtu_record(master_output, filenames);
93 
94  // Associate the time to the time step.
95  _times_filenames.push_back(
96  std::pair<double, std::string>(time, master_filename));
97  }
98 }

110  dealii::LA::distributed::Vector<double> const &enthalpy)
111 {
112  dealii::LA::distributed::Vector<double> temperature(
113  enthalpy.get_partitioner());
114  // This is not used for now because the material properties are independent of
115  // the temperatures.
116  dealii::LA::distributed::Vector<double> state;
117 
118  // TODO the computation does not work if there is a phase change
119  unsigned int const dofs_per_cell = _dof_handler.get_fe().dofs_per_cell;
120  std::vector<dealii::types::global_dof_index> local_dof_indices(dofs_per_cell);
121  for (auto cell :
122  dealii::filter_iterators(_dof_handler.active_cell_iterators(),
123  dealii::IteratorFilters::LocallyOwnedCell()))
124  {
125  double const enthalpy_to_temp =
126  1. / (_material_properties->get(cell, Property::density, state) *
127  _material_properties->get(cell, Property::specific_heat, state));
128 
129  cell->get_dof_indices(local_dof_indices);
130 
131  for (unsigned int i = 0; i < dofs_per_cell; ++i)
132  temperature[local_dof_indices[i]] =
133  enthalpy_to_temp * enthalpy[local_dof_indices[i]];
134  }
135 
136  return temperature;
137 }
