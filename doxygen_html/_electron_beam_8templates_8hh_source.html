53 double ElectronBeam<dim>::value(dealii::Point<dim> const &point,
54  unsigned int const /*component*/) const
55 {
56  double const z = point[1] - _max_height;
57  if ((z + _beam.depth) < 0.)
58  return 0.;
59  else
60  {
61  double const distribution_z =
62  -3. * pow(z / _beam.depth, 2) - 2. * (z / _beam.depth) + 1.;
63 
64  dealii::Point<1> time;
65  time[0] = this->get_time();
66  double const beam_center_x = _position[0].value(time);
67  double xpy_squared = pow(point[0] - beam_center_x, 2);
68  if (dim == 3)
69  {
70  double const beam_center_y = _position[1].value(time);
71  xpy_squared += pow(point[2] - beam_center_y, 2);
72  }
73 
74  double constexpr four_ln_pone = 4. * std::log(0.1);
75  double heat_source = 0.;
76  heat_source =
77  -_beam.energy_conversion_eff * _beam.control_eff * _beam.max_power *
78  four_ln_pone /
79  (dealii::numbers::PI * _beam.diameter_squared * _beam.depth) *
80  std::exp(four_ln_pone * xpy_squared / _beam.diameter_squared) *
81  distribution_z;
