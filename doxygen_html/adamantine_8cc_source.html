138 template <int dim>
139 void run(boost::mpi::communicator const &communicator,
140  std::unique_ptr<adamantine::Physics<dim, double>> &thermal_physics,
141  adamantine::PostProcessor<dim> &post_processor,
142  boost::property_tree::ptree const &time_stepping_database)
143 {
144  thermal_physics->reinit();
145  dealii::LA::distributed::Vector<double> solution;
146  thermal_physics->initialize_dof_vector(solution);
147  unsigned int progress = 1;
148  unsigned int cycle = 0;
149  unsigned int n_time_step = 0;
150  double time = 0.;
151  // Output the initial solution
152  post_processor.output_pvtu(cycle, n_time_step, time, solution);
153  ++n_time_step;
154 
155  double time_step = time_stepping_database.get<double>("time_step");
156  double const duration = time_stepping_database.get<double>("duration");
157  while (time < duration)
158  {
159  if ((time + time_step) > duration)
160  time_step = duration - time;
161  // time can be different than time + time_step if an embedded scheme is
162  // used.
163  time = thermal_physics->evolve_one_time_step(time, time_step, solution);
164 
165  // Get the new time step
166  time_step = thermal_physics->get_delta_t_guess();
167 
168  // Output progress on screen
169  if (communicator.rank() == 0)
170  {
171  double adim_time = time / (duration / 10.);
172  double frac_part = 0;
173  double int_part = 0;
174  frac_part = std::modf(adim_time, &int_part);
175  if ((frac_part < 0.5 * time_step) ||
176  (std::fabs(frac_part - 1.) < 0.5 * time_step))
177  {
178  std::cout << progress * 10 << '%' << " completed" << std::endl;
179  ++progress;
180  }
181  }
182 
183  // Output the solution
184  post_processor.output_pvtu(cycle, n_time_step, time, solution);
185  ++n_time_step;
186  }
187  post_processor.output_pvd();
188 }
