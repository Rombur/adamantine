128 template <int dim>
129 void refine_and_transfer(
130  std::unique_ptr<adamantine::Physics<dim, double>> &thermal_physics,
131  dealii::DoFHandler<dim> &dof_handler,
132  dealii::LA::distributed::Vector<double> &solution)
133 {
134  dealii::parallel::distributed::Triangulation<dim> &triangulation =
135  dynamic_cast<dealii::parallel::distributed::Triangulation<dim> &>(
136  const_cast<dealii::Triangulation<dim> &>(
137  dof_handler.get_triangulation()));
138 
139  std::shared_ptr<adamantine::MaterialProperty<dim>> material_property =
140  thermal_physics->get_material_property();
141 
142  dealii::parallel::distributed::SolutionTransfer<
143  dim, dealii::LA::distributed::Vector<double>>
144  solution_transfer(dof_handler);
145  std::vector<dealii::parallel::distributed::SolutionTransfer<
146  dim, dealii::LA::distributed::Vector<double>>>
147  material_state(static_cast<unsigned int>(adamantine::MaterialState::SIZE),
148  dealii::parallel::distributed::SolutionTransfer<
149  dim, dealii::LA::distributed::Vector<double>>(
150  material_property->get_dof_handler()));
151 
152  // Prepare the Triangulation and the diffent SolutionTransfers for refinement
153  triangulation.prepare_coarsening_and_refinement();
154  solution_transfer.prepare_for_coarsening_and_refinement(solution);
155  for (unsigned int i = 0;
156  i < static_cast<unsigned int>(adamantine::MaterialState::SIZE); ++i)
157  material_state[i].prepare_for_coarsening_and_refinement(
158  material_property->get_state()[i]);
159 
160  // Execute the refinement
161  triangulation.execute_coarsening_and_refinement();
162 
163  // Update the ConstraintMatrix and resize the solution
164  thermal_physics->setup_dofs();
165  thermal_physics->initialize_dof_vector(solution);
166 
167  // Update MaterialProperty DoFHandler and resize the state vectors
168  material_property->reinit_dofs();
169 
170  // Interpolate the solution and the state onto the new mesh
171  solution_transfer.interpolate(solution);
172  for (unsigned int i = 0;
173  i < static_cast<unsigned int>(adamantine::MaterialState::SIZE); ++i)
174  material_state[i].interpolate(material_property->get_state()[i]);
175 
176 #if ADAMANTINE_DEBUG
177  // Check that we are not losing material
178  std::array<dealii::LA::distributed::Vector<double>,
179  static_cast<unsigned int>(adamantine::MaterialState::SIZE)> state =
180  material_property->get_state();
181  unsigned int const local_size = state[0].local_size();
182  unsigned int constexpr n_material_states =
183  static_cast<unsigned int>(adamantine::MaterialState::SIZE);
184  for (unsigned int i = 0; i < local_size; ++i)
185  {
186  double material_ratio = 0.;
187  for (unsigned int j = 0; j < n_material_states; ++j)
188  material_ratio += state[j][i];
189  adamantine::ASSERT(std::abs(material_ratio - 1.) < 1e-14,
190  "Material is lost.");
191  }
192 #endif
193 }

195 template <int dim>
196 std::vector<typename dealii::parallel::distributed::Triangulation<
197  dim>::active_cell_iterator>
198 compute_cells_to_refine(
199  dealii::parallel::distributed::Triangulation<dim> &triangulation,
200  double const time, double const next_refinement_time,
201  unsigned int const n_time_steps,
202  std::vector<std::unique_ptr<adamantine::ElectronBeam<dim>>> &electron_beams)
203 {
204  // Compute the position of the beams between time and next_refinement_time and
205  // refine the mesh where the source is greater than 1e-15. This cut-off is due
206  // to the fact that the source is gaussian and thus never strictly zero. If
207  // the beams intersect, some cells will appear twice in the vector. This is
208  // not a problem.
209  std::vector<typename dealii::parallel::distributed::Triangulation<
210  dim>::active_cell_iterator> cells_to_refine;
211  for (unsigned int i = 0; i < n_time_steps; ++i)
212  {
213  double const current_time = time +
214  static_cast<double>(i) /
215  static_cast<double>(n_time_steps) *
216  (next_refinement_time - time);
217  for (auto &beam : electron_beams)
218  {
219  beam->set_time(current_time);
220  for (auto cell : dealii::filter_iterators(
221  triangulation.active_cell_iterators(),
222  dealii::IteratorFilters::LocallyOwnedCell()))
223  if (beam->value(cell->center()) > 1e-15)
224  cells_to_refine.push_back(cell);
225  }
226  }
227 
228  return cells_to_refine;
229 }

231 template <int dim, int fe_degree>
232 void refine_mesh(
233  std::unique_ptr<adamantine::Physics<dim, double>> &thermal_physics,
234  dealii::LA::distributed::Vector<double> &solution,
235  std::vector<std::unique_ptr<adamantine::ElectronBeam<dim>>> &electron_beams,
236  double const time, double const next_refinement_time,
237  unsigned int const time_steps_refinement,
238  boost::property_tree::ptree const &refinement_database)
239 {
240  dealii::DoFHandler<dim> &dof_handler = thermal_physics->get_dof_handler();
241  // Use the Kelly error estimator to refine the mesh. This is done so that the
242  // part of the domain that were heated stay refined.
243  unsigned int const n_kelly_refinements =
244  refinement_database.get("n_heat_refinements", 2);
245  double coarsening_fraction = 0.3;
246  double refining_fraction = 0.6;
247  double cells_fraction = refinement_database.get("heat_cell_ratio", 1.);
248  dealii::parallel::distributed::Triangulation<dim> &triangulation =
249  dynamic_cast<dealii::parallel::distributed::Triangulation<dim> &>(
250  const_cast<dealii::Triangulation<dim> &>(
251  dof_handler.get_triangulation()));
252  // Number of times the mesh on the beam paths will be refined and maximum
253  // number of time a cell can be refined.
254  unsigned int const n_beam_refinements =
255  refinement_database.get("n_beam_refinements", 2);
256  int max_level = refinement_database.get<int>("max_level");
257  for (unsigned int i = 0; i < n_kelly_refinements; ++i)
258  {
259  // Estimate the error. For simplicity, always use dealii::QGauss
260  dealii::Vector<float> estimated_error_per_cell(
261  triangulation.n_active_cells());
262  dealii::KellyErrorEstimator<dim>::estimate(
263  dof_handler, dealii::QGauss<dim - 1>(fe_degree + 1),
264  typename dealii::FunctionMap<dim>::type(), solution,
265  estimated_error_per_cell, dealii::ComponentMask(), nullptr, 0,
266  triangulation.locally_owned_subdomain());
267 
268  // Flag the cells for refinement.
269  unsigned int new_n_cells = static_cast<unsigned int>(
270  cells_fraction *
271  static_cast<double>(triangulation.n_global_active_cells()));
272  dealii::GridRefinement::refine_and_coarsen_fixed_fraction(
273  triangulation, estimated_error_per_cell, refining_fraction,
274  coarsening_fraction, new_n_cells);
275 
276  // Don't refine cells that are already as much refined as it is allowed.
277  for (auto cell :
278  dealii::filter_iterators(triangulation.active_cell_iterators(),
279  dealii::IteratorFilters::LocallyOwnedCell()))
280  if (cell->level() >= max_level)
281  cell->clear_refine_flag();
282 
283  // Execute the refinement and transfer the solution onto the new mesh.
284  refine_and_transfer(thermal_physics, dof_handler, solution);
285  }
286 
287  // Refine the mesh along the trajectory of the sources.
288  for (unsigned int i = 0; i < n_beam_refinements; ++i)
289  {
290  // Compute the cells to be refined.
291  std::vector<typename dealii::parallel::distributed::Triangulation<
292  dim>::active_cell_iterator> cells_to_refine =
293  compute_cells_to_refine(triangulation, time, next_refinement_time,
294  time_steps_refinement, electron_beams);
295 
296  // Flag the cells for refinement.
297  for (auto &cell : cells_to_refine)
298  if (cell->level() < max_level)
299  cell->set_refine_flag();
300 
301  // Execute the refinement and transfer the solution onto the new mesh.
302  refine_and_transfer(thermal_physics, dof_handler, solution);
303  }
304 
305  // Recompute the inverse of the mass matrix and update the material
306  // properties.
307  thermal_physics->reinit();
308 }

399 template <int dim>
400 void run(
401  boost::mpi::communicator const &communicator,
402  std::unique_ptr<adamantine::Physics<dim, double>> &thermal_physics,
403  adamantine::PostProcessor<dim> &post_processor,
404  std::vector<std::unique_ptr<adamantine::ElectronBeam<dim>>> &electron_beams,
405  boost::property_tree::ptree const &time_stepping_database,
406  boost::property_tree::ptree const &refinement_database,
407  unsigned int const fe_degree, double const initial_temperature)
408 {
409  thermal_physics->setup_dofs();
410  thermal_physics->reinit();
411  dealii::LA::distributed::Vector<double> solution;
412  thermal_physics->initialize_dof_vector(initial_temperature, solution);
413  unsigned int progress = 1;
414  unsigned int cycle = 0;
415  unsigned int n_time_step = 0;
416  double time = 0.;
417  // Output the initial solution
418  dealii::ConstraintMatrix &constraint_matrix =
419  thermal_physics->get_constraint_matrix();
420  constraint_matrix.distribute(solution);
421  post_processor.output_pvtu(cycle, n_time_step, time, solution);
422  ++n_time_step;
423 
424  bool const verbose_refinement = refinement_database.get("verbose", false);
425  unsigned int const time_steps_refinement =
426  refinement_database.get("time_steps_between_refinement", 10);
427  double next_refinement_time = time;
428  double time_step = time_stepping_database.get<double>("time_step");
429  double const duration = time_stepping_database.get<double>("duration");
430  while (time < duration)
431  {
432  if ((time + time_step) > duration)
433  time_step = duration - time;
434 
435  // Refine the mesh after time_steps_refinement time steps or when time is
436  // greater or equal than the next predicted time for refinement. This is
437  // necessary when using an embedded method.
438  if (((n_time_step % time_steps_refinement) == 0) ||
439  (time >= next_refinement_time))
440  {
441  next_refinement_time = time + time_steps_refinement * time_step;
442  refine_mesh(thermal_physics, solution, electron_beams, time,
443  next_refinement_time, time_steps_refinement,
444  refinement_database, fe_degree);
445  if ((communicator.rank() == 0) && (verbose_refinement == true))
446  std::cout << "n_dofs: " << thermal_physics->get_dof_handler().n_dofs()
447  << std::endl;
448  }
449 
450  // time can be different than time + time_step if an embedded scheme is
451  // used.
452  time = thermal_physics->evolve_one_time_step(time, time_step, solution);
453 
454  // Get the new time step
455  time_step = thermal_physics->get_delta_t_guess();
456 
457  // Output progress on screen
458  if (communicator.rank() == 0)
459  {
460  double adim_time = time / (duration / 10.);
461  double frac_part = 0;
462  double int_part = 0;
463  frac_part = std::modf(adim_time, &int_part);
464  if ((frac_part < 0.5 * time_step) ||
465  (std::fabs(frac_part - 1.) < 0.5 * time_step))
466  {
467  std::cout << progress * 10 << '%' << " completed" << std::endl;
468  ++progress;
469  }
470  }
471 
472  // Output the solution
473  constraint_matrix.distribute(solution);
474  post_processor.output_pvtu(cycle, n_time_step, time, solution);
475  ++n_time_step;
476  }
477  post_processor.output_pvd();
478 }
